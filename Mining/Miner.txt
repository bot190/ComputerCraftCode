-- Create reference to modem
modem = peripheral.wrap("right")
modem.open(192)

end

function move()
	if turtle.back() == false then
		turtle.turnRight()
		turtle.turnRight()
		turtle.dig()
		turtle.up()
		turtle.dig()
		turtle.down()
		turtle.turnRight()
		turtle.turnRight()
		turtle.back()
	end
end

function setup()
	--Conduit in slot 1
	turtle.select(1)
	turtle.place()
	turtle.back()
	--Mining Well in slot 2
	turtle.select(2)
	turtle.place()
	turtle.up()
	--EnderChest in slot 3
	turtle.select(3)
	turtle.place()
	modem.transmit(192,192, "Done")
end


while true do
	local event, modemSide, senderChannel, replyChannel, message, senderDistance = os.pullEvent("modem_message")
	if (message == "Pack") then
		pack()
	elseif message == "Move" then
		move()
	elseif message == "Setup" then
		setup()
	end
end
